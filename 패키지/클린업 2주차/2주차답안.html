set.seed(1234)
hold_out_index <- createDataPartition(data$AMT, p = 0.3, list = F)
train_h <- data[-hold_out_index,]
test_h <- data[hold_out_index,]



ho_result <- data.frame(variable = rep(NA,ncol(combination)),
                        RMSE = rep(NA, ncol(combination)))

pb <- progress_bar$new(total = ncol(combination))
for(i in 1:ncol(combination)){
  pb$tick()
  ho_result[i,"variable"] = paste0("Combination_", i)
  model = lm(AMT ~., data = select(train_h, -c(combination[,i])))
  temp_pred = predict(model, newdata = select(test_h, -c(AMT, combination[,i])))
  temp_RMSE = MLmetrics::RMSE(temp_pred, test_h$AMT)
  ho_result[i,"RMSE"] = temp_RMSE
}

cv_result <- data.frame("variable" = rep(NA,ncol(combination)),
                        "RMSE" = rep(NA, ncol(combination)))


set.seed(1234)
cv_index <- createFolds(data$AMT, k = 5)


pb <- progress_bar$new(total = ncol(combination))
for(i in 1:ncol(combination)){
  pb$tick()
  cv_temp = c()
  cv_result[i,"variable"] = paste0("Combination_", i)
  for(k in 1:5){
    cv_train = data[-cv_index[[k]],]
    cv_test = data[cv_index[[k]],]
    model = lm(AMT ~., data = select(cv_train, -c(combination[,i])))
    temp_pred = predict(model, newdata = select(cv_test, -c(AMT, combination[,i])))
    temp_RMSE = MLmetrics::RMSE(temp_pred, cv_test$AMT)
    
    cv_temp[k] = temp_RMSE
    }
  cv_result[i,"RMSE"] = mean(cv_temp)
}

2-7번

  • Training RMSE 보다 Test RMSE 크게 나왔다
  • 당연히 모델 자체가 Training data를 가지고 변수선택(튜닝)했기 때문
  • Hold out method 보다 CV method의 결과가 TEST RMSE가 낮게 나왔다
  • Hold out method 보다 CV method가 보다 잘 일반화 되어 있기 때문
