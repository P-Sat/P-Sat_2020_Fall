getmode <- function(v) {
  uniqv <- unique(v)
  if(sum(is.na(uniqv)) == 1){
    uniqv = uniqv[-which(is.na(uniqv))]
    }
  uniqv[which.max(tabulate(match(v, uniqv)))]
}

train_mode[which(is.na(train_mode$CSTMR_CNT)), "CSTMR_CNT"] = getmode(train_mode$CSTMR_CNT)

평균 설명
  • 평균 대체의 경우 “na.rm = T” 로 인해 같게 나옴.
  • 랜덤 대체의 경우 랜덤으로 인해 seed 값에 따라 다르게 나옴.
