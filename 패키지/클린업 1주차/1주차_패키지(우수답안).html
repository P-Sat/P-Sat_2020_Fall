2-4번. train_mode의 CSTMR_CNT변수의 NA값들을 최빈값 대체대체(Mode imputation) 방법으로 대체해주세요.

getmode <- function(v) {
  uniqv <- unique(v)
  if( sum(is.na(uniqv)) == 1){
    uniqv = uniqv[-which(is.na(uniqv))]
  }
  uniqv[which.max(tabulate(match(v, uniqv)))]
}

train_mode[is.na(train_mode$CSTMR_CNT==T),"CSTMR_CNT"] = getmode(train_mode$CSTMR_CNT)
