best_param <- list()
best_seed.num <- 1234
best_rmse <- Inf

pb <- progress_bar$new(total = 10)
for(iter in 1:10){
  pb$tick()
  #print(paste0("iteration-",iter))
  
  param <- list(
    objective = "reg:squarederror",
    eval_metric = "rmse",
    max_depth = sample(3:8, 1),
    min_child_weight = sample(3:8, 1),
    subsample = runif(1, 0.6, 1),
    colsample_bytree = runif(1, 0.6, 1),
    eta = runif(1, 0.01, 0.3)
    )
  
  cv_nround = sample(seq(from = 500, to = 800, by = 100), 1)
  seed_num <- sample.int(10000, 1)[[1]]
  
  
  cv_rmse = c()
  for(k in 1:5){
    #print(paste0(k, "-fold"))
    cv_train = train_dummy[-cv_index[[k]], ]
    cv_test = train_dummy[cv_index[[k]], ]
    cv_tr_label = cv_train$AMT
    cv_te_label = cv_test$AMT
    xgb_cv_tr = xgb.DMatrix(as.matrix(select(cv_train, -AMT)), label = cv_tr_label)
    xgb_cv_te = xgb.DMatrix(as.matrix(select(cv_test, -AMT)), label = cv_te_label)
    cv_watchlist = list(msr = xgb_cv_tr)
    
    set.seed(seed_num)
    xgb_cv_model <- xgb.train(params = param,
                              data = xgb_cv_tr,
                              nrounds = cv_nround,
                              watchlist = cv_watchlist,
                              verbose = F,
                              print_every_n = 100,
                              early_stopping_rounds = 0.05*cv_nround)
    temp_pred = predict(xgb_cv_model, xgb_cv_te)
    temp_RMSE = RMSE(temp_pred, cv_te_label)
    cv_rmse[k] = temp_RMSE
  }
  min_rmse = mean(cv_rmse)
  min_rmse_round = cv_nround
  
  if(min_rmse < best_rmse){
    best_rmse = min_rmse
    best_rmse_round = min_rmse_round
    best_seed.num = seed_num
    best_param = param
  }
}
